\subsubsection{A Gift For You}
Ephesians 2:1-10:\footnote{06 September 2014, Stephen Guthrie}
\index[speaker]{Stephen Guthrie!Ephesians 2:01-10 (A Gift For You)}
\index[series]{Ephesians (Stephen Guthrie)!Ephesians 2:01-10 (A Gift For You)}
\index[date]{2014/09/06!Ephesians 2:1-10 (A Gift For You) (Stephen Guthrie)}
\begin{compactenum}[I.]
    \item \textbf{The CONTENT of the gift} \index[scripture]{Ephesians!Ephesians 02:08} (Ephesians 2:8)
    \begin{compactenum}[A.]
    	\item This gift is salvation.
    	\begin{compactenum}[1.]
    		\item Salvation is forgiveness of sins. \index[scripture]{Ephesians!Ephesians 01:07} (Ephesians 1:7)
    		\item Salvation is freedom from sin. \index[scripture]{Galatians!Galatians 05:01} (Galatians 5:1)
    		\item Salvation is forever. (\index[scripture]{1 Timothy!1 Timothy 01:12}1 Timothy 1:12, \index[scripture]{Romans!Romans 06:23} Romans 6:23)
    		\item Salvation is having a fortress to run to. \index[scripture]{Psalms!Psalm 18:02} (Psalm 18:2)
    		\item Salvation places us in a new family; God's family.
    		\item Salvation gives us a function. \index[scripture]{Ephesians!Ephesians 02:10} (Ephesians 2:10)
    		\item Salvation is for everyone. \index[scripture]{Titus!Titus 02:11}  (Titus 2:11)
    		\item Salvation is our flight from wrath. \index[scripture]{Ephesians!Ephesians 05:09} (Ephesians 5:9)
    		\begin{compactenum}[i.]
    			\item Hell
    			\item Tribulation Period
    		\end{compactenum}
    	\end{compactenum}
	\end{compactenum}
    \item \textbf{The COST of the gift} \index[scripture]{John!John 19:30} (John 19:30)
    \begin{compactenum}[A.]
    	\item This gift comes through faith.
    	    \begin{compactenum}[1.]
		    	\item Jesus paid for this gift with his life.
		    	\item There is no cost to us for salvation, a gift is free to the person receiving it.
		    \end{compactenum}
	\end{compactenum}
	\item \textbf{The COVVEYING of the gift} \index[scripture]{Ephesians!Ephesians 02:08} (Ephesians 2:8)
    \begin{compactenum}[A.]
    	\item This gift comes by grace.
    	\begin{compactenum}[1.]
    		\item Grace is stronger than any sin that you and I have ever committed. \index[scripture]{Romans!Romans 05:20}  (Romans 5:20)
    	\end{compactenum}
	\end{compactenum}
    \item \textbf{The COLLECTING of the gift} \index[scripture]{Ephesians!Ephesians 02:08} (Ephesians 2:8)
    \begin{compactenum}[A.]
    	\item This gift comes through faith.
    	    \begin{compactenum}[1.]
		    	\item Jesus gives us this measure of faith. \index[scripture]{Romans!Romans 12:03} (Romans 12:3)
		    	\item We put our measure of faith in Jesus where we call on him. \index[scripture]{Romans!Romans 10:09-13} (Romans 10:9-13)
		    \end{compactenum}
    	\item We do not collect this gift (get saved) by our moral lifestyle, our righteous works, or our financial status, or by the way that we left this life.
    \end{compactenum}
\end{compactenum}
\textbf{Conclusion: } \index[scripture]{Romans!Romans 06:13} Romans 6:23
